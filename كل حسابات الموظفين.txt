SELECT     dbo.TblEmployee.Fullcode, dbo.TblEmployee.Emp_Name, dbo.ACCOUNTS.Account_Serial AS [حساب الذمة], dbo.ACCOUNTS.Account_Name AS [حساب الذمة], 
                      ACCOUNTS_1.Account_Serial AS [اجور مستحقة], ACCOUNTS_1.Account_Name AS [اجور مستحقة], ACCOUNTS_2.Account_Serial AS [مخصص اجازة], 
                      ACCOUNTS_2.Account_Name AS [مخصص اجازة], ACCOUNTS_3.Account_Name AS [مدفوعات مقدمة], ACCOUNTS_3.Account_Serial AS [مدفوعات مقدمة], 
                      ACCOUNTS_4.Account_Name AS [مخصص نهاية خدمة], ACCOUNTS_4.Account_Serial AS [مخصص نهاية خدمة], ACCOUNTS_5.Account_Name AS [مخصص تذاكر], 
                      ACCOUNTS_5.Account_Serial AS [مخصص تذاكر]
FROM         dbo.ACCOUNTS INNER JOIN
                      dbo.TblEmployee ON dbo.ACCOUNTS.Account_Code = dbo.TblEmployee.Account_code INNER JOIN
                      dbo.ACCOUNTS ACCOUNTS_1 ON dbo.TblEmployee.Account_code1 = ACCOUNTS_1.Account_Code INNER JOIN
                      dbo.ACCOUNTS ACCOUNTS_2 ON dbo.TblEmployee.Account_Code2 = ACCOUNTS_2.Account_Code INNER JOIN
                      dbo.ACCOUNTS ACCOUNTS_3 ON dbo.TblEmployee.Account_Code3 = ACCOUNTS_3.Account_Code INNER JOIN
                      dbo.ACCOUNTS ACCOUNTS_4 ON dbo.TblEmployee.Account_Code4 = ACCOUNTS_4.Account_Code INNER JOIN
                      dbo.ACCOUNTS ACCOUNTS_5 ON dbo.TblEmployee.Account_Code5 = ACCOUNTS_5.Account_Code
